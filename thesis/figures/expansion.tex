\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta}

\begin{document}
\begin{tikzpicture}[>=Stealth]
	\coordinate (O) at (0,0);
	\draw[gray] (-5,0) -- (O);
	\draw[gray,->] (O) -- (5,0) node[above]{$x$}; % x-axis
	\draw[gray,->] (O) -- (0,5) node[left ]{$t$}; % t-axis
	% constant region (left)
	\foreach \x in {-0.6,-1.2,...,-3.6}   \draw[->] (\x,0) -- (\x-1,4);
	% constant region (right)
	\foreach \x in {+0.6,+1.2,...,+3.0}   \draw[->] (\x,0) -- (\x+2,4);
	% the expansion region
	\foreach \x in {-1,-0.4,...,+2}   \draw[->, very thick] (O) -- (\x,4); 
	% wave head and tail
	\draw[->, very thick] (O) -- (-1,4) node[above]{$\lambda_{-}$}; 
	\draw[->, very thick] (O) -- (+2,4) node[above]{$\lambda_{+}$}; 
\end{tikzpicture}
\end{document}
